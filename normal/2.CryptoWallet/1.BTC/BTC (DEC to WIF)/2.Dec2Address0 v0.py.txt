# Dec2Address v0
# dec.txt поместить в папку с Python, после генерации появится adresses.txt
# В dec.txt последняя строка должна быть пустой, иначе съедет вывод!!! 


from bitcoin import *

file = open("2.Dec2Address1 v0(dec).txt", "r")
out = open("2.Dec2Address2 v0(adresses).txt", "w")
addresses = file.readlines()
for address in addresses:
    try:
        priv = hex(int(address))[2:].zfill(64)
    except:
        continue
    pub = privtopub(priv)
    addr = pubtoaddr(pub)

    print(f"  DEC   : {address}"
            f"PrivKey : {priv}\n"
            f"PubKey  : {pub}\n"
            f"Address : {addr}\n\n")

    out.write(f"  DEC   : {address}"
            f"PrivKey : {priv}\n"
            f"PubKey  : {pub}\n"
            f"Address : {addr}\n\n")

out.close()
