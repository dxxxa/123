#Код для персонализации адреса биткойнов в Python

from bitcoin import *   # need to import https://pypi.org/project/bitcoin/  a very clean lib from Vitalik Butarin

print('Starting scan...')
Yourname='1FeexV6bAHb8ybZjqQMjJrcCrHGW9sb6uF'

while True:

    priv=random_key()
    pub=privtopub(priv)
    addr=pubtoaddr(pub)
    print("priv:" + str(priv))
    print("pub:" + str(pub))
    print("addr:" + str(addr))
    if pub[1:len(Yourname)]=='1FeexV6bAHb8ybZjqQMjJrcCrHGW9sb6uF':   #Find a btc address with "Yourname" eg.: 1Yournamexxxxxxxxxxx
            print(priv,pub,addr)
