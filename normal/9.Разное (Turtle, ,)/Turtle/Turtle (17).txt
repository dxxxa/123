import turtle as t
t.bgcolor("black")
color = ("darkolivegreen","mediumvioletred")
for i in range(130):
    t.speed(0)
    t.pencolor(color[i%2])
    t.fillcolor(color[i%2])
    t.speed(0)
    t.begin_fill()
    t.forward(i)
    t.right(16)
    t.circle(30,20)
    t.forward(i)
    t.right(30)
    t.left(30)
    t.circle(20,30)
    t.right(30)
    t.backward(i)
    t.left(400)
    t.end_fill()
t.done()
