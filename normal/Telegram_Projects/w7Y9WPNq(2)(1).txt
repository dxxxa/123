from pyrogram import Client  # —Ç–µ–ª–µ–≥—Ä–∞–º –∫–ª–∏–µ–Ω—Ç
import shelve  # –∑–∞–ø–∏—Å—å –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–∏ –≤ —Ñ–∞–π–ª
import random  # –¥–ª—è –≤—ã–±–æ—Ä–∞ —Å–ª—É—á–∞–π–Ω–æ–≥–æ —ç–ª–µ–º–µ–Ω–∏—Ç–∞ –∏–∑ —Å–ø–∏—Å–∫–∞
import time  # –¥–ª—è –≤—ã–±–æ—Ä–∞ —Å–ª—É—á–∞–π–Ω–æ–≥–æ —ç–ª–µ–º–µ–Ω–∏—Ç–∞ –∏–∑ —Å–ø–∏—Å–∫–∞


# –î–∞–Ω–Ω—ã–µ –ø—Ä–∏–ª–æ–∂–µ–Ω–∏—è —Ç–µ–ª–µ–≥—Ä–∞–º (–º–æ–∂–Ω–æ —Å–æ–∑–¥–∞—Ç—å –Ω–∞ my.telegram.org)
api_id = 183162
api_hash = "d89672a589c14e14129a713a01c86d0c"
phone_number = '+........'  # —Ç–µ–ª–µ—Ñ–æ–Ω–Ω—ã–π –Ω–æ–º–µ—Ä –∞–∫–∫–∞—É–Ω—Ç–∞ –≤ —Ç–µ–ª–µ–≥—Ä–∞–º–µ
PUBLIC = 'tupogram'  # –ø–∞–±–ª–∏–∫
# –í–∞—Ä–∏–∞–Ω—Ç—ã —Ç–µ–∫—Å—Ç–æ–≤ —Å–æ–æ–±—â–µ–Ω–∏–π
TEXTS = [
    '–ö—Ä—É—Ç–æ !',
    '—Ç–æ–∂–µ —Ç–∞–∫ —Ö–æ—á—É üòé',
    '5 –±–∞–ª–ª–æ–≤ üëç',
]
COMMENT_EVERY_N = 1  # –∫–æ–º–º–µ–Ω—Ç–∏—Ä—É–µ–º –∫–∞–∂–¥–æ–µ N —Å–æ–±—â–µ–Ω–∏–µ
                     # –µ—Å–ª–∏ —Ä–∞–≤–Ω–æ 1, –∫–æ–º–º–µ–Ω—Ç–∏—Ä—É–µ–º –∫–∞–∂–¥–æ–µ —Å–æ–æ–±—â–µ–Ω–∏–µ
                     # –µ—Å–ª–∏ —Ä–∞–≤–Ω–æ 3, –∫–æ–º–º–µ–Ω—Ç–∏—Ä—É–µ–º –∫–∞–∂–¥–æ–µ —Ç—Ä–µ—Ç—å–µ
                     # –µ—Å–ª–∏ —Ä–∞–≤–Ω–æ 5, –∫–æ–º–º–µ–Ω—Ç–∏—Ä—É–µ–º –∫–∞–∂–¥–æ–µ –ø—è—Ç–æ–µ ... –∏—Ç–¥

# —Å–ø–∏—Å–æ–∫ –æ–±—Ä–∞–±–æ—Ç–∞–Ω–Ω—ã—Ö —Å–æ–æ–±—â–µ–Ω–∏–π
processed_messages = shelve.open('processed_messages.db',  # –∏–º—è —Ñ–∞–π–ª–∞, –∫—É–¥–∞ –ø–∏—Å–∞—Ç—å
                                 writeback=True)

# —Å–æ–∑–¥–∞–µ–º –∫–ª–∏–µ–Ω—Ç —Ç–µ–ª–µ–≥—Ä–∞–º–∞
app = Client("python_cyberpunk", api_id, api_hash,
             phone_number=phone_number)

with app:
    public = app.get_chat(PUBLIC)  # –∏—â–µ–º –ø–∞–±–ª–∏–∫ –ø–æ –Ω–∏–∫—É
    chat = public.linked_chat  # —Å–≤—è–∑–∞–Ω–Ω—ã–π —á–∞—Ç –æ–±—Å—É–∂–¥–µ–Ω–∏–π –ø–∞–±–ª–∏–∫–∞
    
    while True:
        for msg in app.get_history(chat.id, limit=100):
            # —Ñ–∏–ª—å—Ç—Ä—É–µ–º —Ç–æ–ª—å–∫–æ –∞–≤—Ç–æ—Ä–µ–ø–æ—Å—Ç—ã –∏–∑ –ø–∞–±–ª–∏–∫–∞
            if (msg.from_user is None  # –µ—Å–ª–∏ —Å–æ–æ–±—â–µ–Ω–∏–µ –Ω–µ –∏–º–µ–µ—Ç –∞–≤—Ç–æ—Ä–∞
                    # –∏ —ç—Ç–æ —Ä–µ–ø–æ—Å—Ç –∏–∑ –ø–∞–±–ª–∏–∫–∞ (–ø—Ä–æ–≤–µ—Ä–∫–∞ –ø–æ id)
                    and msg.forward_from_chat.id == public.id):
                if msg.forward_from_message_id % COMMENT_EVERY_N != 0:
                    print(f'–ü—Ä–æ–ø—É—Å–∫–∞–µ–º message_id={msg.message_id},'
                        f' —Ç–∞–∫ –∫–∞–∫ –∫–æ–º–º–µ–Ω—Ç–∏—Ä—É–µ –∫–∞–∂–¥–æ–µ {COMMENT_EVERY_N}')
                    continue
                # –ø—Ä–æ–≤–µ—Ä—è–µ–º, –µ—Å—Ç—å –ª–∏ –≤ —Å–ø–∏—Å–∫–µ –æ–±—Ä–∞–±–æ—Ç–∞–Ω–Ω—ã—Ö —Å–æ–æ–±—â–µ–Ω–∏–π —ç—Ç–æ—Ç –∞–π–¥–∏
                # —á—Ç–æ–±—ã –Ω–µ –∫–æ–º–º–µ–Ω—Ç–∏—Ä–æ–≤–∞—Ç—å –ø–æ –Ω–µ—Å–∫–æ–ª—å–∫–æ —Ä–∞–∑ –æ–¥–∏–Ω –ø–æ—Å—Ç
                if str(msg.forward_from_message_id) in processed_messages:
                    print(f'–ü—Ä–æ–ø—É—Å–∫–∞–µ–º —É–∂–µ –æ–±—Ä–∞–±–æ—Ç–∞–Ω–Ω–æ–µ message_id={msg.message_id}')
                    continue
                # –ø–∏—à–µ–º –≤ —Å–ø–∏—Å–æ–∫ –æ–±—Ä–∞–±–æ—Ç–∞–Ω–Ω—ã—Ö –∞–π–¥–∏ —ç—Ç–æ–≥–æ —Å–æ–æ–±—â–µ–Ω–∏—è
                processed_messages[str(msg.forward_from_message_id)] = True

                print(f'–û–±—Ä–∞–±–æ—Ç–∫–∞ message_id={msg.message_id}')

                text = random.choice(TEXTS)  # –≤—ã–±–∏—Ä–∞–µ–º —Å–ª—É—á–∞–π–Ω—ã–π —Ç–µ–∫—Å—Ç –∏–∑ —Å–ø–∏—Å–∫–∞
                app.send_message(chat.id, text,  # –æ—Ç–ø—Ä–∞–≤–ª—è–µ–º —Ç–µ–∫—Å—Ç –≤ —á–∞—Ç
                                reply_to_message_id=msg.message_id)  # –∫–∞–∫ –æ—Ç–≤–µ—Ç –Ω–∞ —Å–æ–æ–±—â–µ–Ω–∏–µ —Å –ø–æ—Å—Ç–æ–º
                
                # –¥–ª—è —Ç–æ–≥–æ, —á—Ç–æ–± –Ω–µ –æ—Å—Ç–∞–≤–ª—è—Ç—å –±–æ–ª—å—à–µ –æ–¥–Ω–æ–≥–æ –∫–æ–º–º–µ–Ω—Ç–∞ –∑–∞ 5 –º–∏–Ω—É—Ç
                break  # –≤—ã—Ö–æ–¥–∏–º –∏–∑ –ø–µ—Ä–µ–±–æ—Ä–∞ —Å–æ–æ–±—â–µ–Ω–∏–π, –µ—Å–ª–∏ –æ—Å—Ç–∞–≤–∏–ª–∏ –∫–æ–º–º–µ–Ω—Ç

        print('–°—Ç–∞–≤–∏–º –Ω–∞ –ø–∞—É–∑—É')
        time.sleep(60 * 5)
