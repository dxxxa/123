import random
import time

t1 = time.time()
s = input()
l = []
a = [[random.randint(0, 1) for i in range(len(s))] for j in range(len(s) * 1000)]
for i in a:
    n = list(s)
    for j in range(len(i)):
        n[j] = n[j].lower() if not i[j] else n[j].upper()
    l.append("".join(n))

print(*sorted(set(l))[::-1], sep="\n")
print(f"Количество вариантов: {len(set(l))}. " + ("Совпадает" if 2 ** len(s) == len(set(l)) else "Не совпадает"))
t2 = time.time()
print(t2 - t1)
